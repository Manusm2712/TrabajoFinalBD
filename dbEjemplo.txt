CREATE TABLE cliente(
	id NUMERIC(10) PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL
);

CREATE TABLE cupon(
	codigo NUMBER(10) PRIMARY KEY , 
	valor_descuento NUMBER(6) NOT NULL,
	fecha_emision DATE NOT NULL,
	fecha_vencimiento DATE NOT NULL,
	id_cliente NUMBER(10) REFERENCES cliente
);

CREATE TABLE servicio(
	nombre VARCHAR(10) PRIMARY KEY,
	duracion VARCHAR(10) NOT NULL,
	precio NUMERIC(6) NOT NULL,
	nombre_servicio VARCHAR(10) REFERENCES servicio(nombre)
);
